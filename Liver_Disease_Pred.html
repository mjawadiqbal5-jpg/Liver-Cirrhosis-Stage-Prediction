# Clean and Normalise the dataset

# Drop the 'ID' column as it has no predictive power
if 'ID' in df.columns:
    df = df.drop('ID', axis=1)

# Drop rows where the target variable ('Stage') is missing
df = df.dropna(subset=['Stage']).copy()

# Identify categorical columns and target
cat_cols = df.select_dtypes(include=['object']).columns
num_cols = df.select_dtypes(include=['float64', 'int64']).columns.drop('Stage')

# Impute categorical columns with the Mode (most frequent value)
for col in cat_cols:
    df[col] = df[col].fillna(df[col].mode()[0])

# Seperate the X and y

# One-Hot Encode categorical variables

# XGBoost requires class labels to start at 0 (e.g., Stage 1.0-4.0 becomes 0-3)

# Training the model

for fold, (train_index, test_index) in enumerate(skf.split(X, y), 1):
    # Split the data first
    X_train, X_test = X.iloc[train_index].copy(), X.iloc[test_index].copy()
    y_train, y_test = y[train_index], y[test_index]

    for col in num_cols:
        # 1. Impute using Train Median
        train_median = X_train[col].median()
        X_train[col] = X_train[col].fillna(train_median)
        X_test[col]  = X_test[col].fillna(train_median)

        # 2. Cap Outliers using Train IQR
        Q1 = X_train[col].quantile(0.25)
        Q3 = X_train[col].quantile(0.75)
        IQR = Q3 - Q1
        lower_bound = Q1 - 1.5 * IQR
        upper_bound = Q3 + 1.5 * IQR

        X_train[col] = np.clip(X_train[col], lower_bound, upper_bound)
        X_test[col]  = np.clip(X_test[col], lower_bound, upper_bound)

    # --- Tune & Train Random Forest ---
    print("    Tuning Random Forest...")
    rf_search = RandomizedSearchCV(base_rf, rf_params, n_iter=10, cv=3, random_state=42, n_jobs=-1, verbose=0)
    rf_search.fit(X_train, y_train)
    best_rf = rf_search.best_estimator_ # Grab the winning model!

    rf_feature_importances += best_rf.feature_importances_ / skf.n_splits
    rf_preds = best_rf.predict(X_test)

    results.append({'Model': 'Random Forest', 'Metric': 'Accuracy', 'Score': accuracy_score(y_test, rf_preds)})
    results.append({'Model': 'Random Forest', 'Metric': 'Precision', 'Score': precision_score(y_test, rf_preds, average='weighted', zero_division=0)})
    results.append({'Model': 'Random Forest', 'Metric': 'Recall', 'Score': recall_score(y_test, rf_preds, average='weighted', zero_division=0)})

    # --- Tune & Train XGBoost ---
    print("    Tuning XGBoost...")
    xgb_search = RandomizedSearchCV(base_xgb, xgb_params, n_iter=10, cv=3, random_state=42, n_jobs=-1, verbose=0)
    xgb_search.fit(X_train, y_train)
    best_xgb = xgb_search.best_estimator_ # Grab the winning model!

    xgb_feature_importances += best_xgb.feature_importances_ / skf.n_splits
    xgb_preds = best_xgb.predict(X_test)

    results.append({'Model': 'XGBoost', 'Metric': 'Accuracy', 'Score': accuracy_score(y_test, xgb_preds)})
    results.append({'Model': 'XGBoost', 'Metric': 'Precision', 'Score': precision_score(y_test, xgb_preds, average='weighted', zero_division=0)})
    results.append({'Model': 'XGBoost', 'Metric': 'Recall', 'Score': recall_score(y_test, xgb_preds, average='weighted', zero_division=0)})

    # Print the detailed classification report for the first fold
    if fold == 1:
        target_names = ['Stage 1', 'Stage 2', 'Stage 3', 'Stage 4']
        print("--- Fold 1: Random Forest Classification Report ---")
        print(classification_report(y_test, rf_preds, target_names=target_names, zero_division=0))
        print("\n--- Fold 1: XGBoost Classification Report ---")
        print(classification_report(y_test, xgb_preds, target_names=target_names, zero_division=0))

# Print training/test set sizes and distributions (for the last fold)
print(f"Training set size:  {X_train.shape[0]} samples")
print(f"Test set size:      {X_test.shape[0]} samples")
print(f"\nTraining class distribution:\n{pd.Series(y_train).value_counts()}")
print(f"\nTest class distribution:\n{pd.Series(y_test).value_counts()}")
